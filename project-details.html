1. API Integration

I have strong expertise in integrating RESTful APIs and third-party services to efficiently connect front-end applications with backend systems.

  • Experience with REST and JSON APIs
  • Connecting front-end with backend data
  • Secure and optimized API handling
import React, { useEffect, useState } from 'react';

function UsersList() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('https://jsonplaceholder.typicode.com/users')
      .then(response => response.json())
      .then(data => {
        setUsers(data);
        setLoading(false);
      })
      .catch(error => console.error('Error:', error));
  }, []);

  if (loading) return <p>Loading users...</p>;

  return (
    <ul>
      {users.map(user => (<li key={user.id}>{user.name} - {user.email}</li>))}
    </ul>
  );
}

2. React Hooks

I effectively use React Hooks to manage component state and lifecycle logic, ensuring cleaner and more efficient functional components.

  • Proficient with useState and useEffect
  • Reusable custom hook creation
  • Improved code readability & performance
import React, { useState, useEffect } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    document.title = `Count: ${count}`;
  }, [count]);

  return (
    <div>
      <p>Current count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}

3. Context API

I utilize the Context API to manage and share state globally across components, reducing prop drilling and improving scalability.

  • Efficient global state management
  • Clean component communication
  • Lightweight alternative to Redux
import React, { createContext, useContext, useState } from 'react';

const ThemeContext = createContext();

function App() {
  const [theme, setTheme] = useState('light');

  return (
    <ThemeContext.Provider value={{ theme, setTheme }}>
      <Toolbar />
    </ThemeContext.Provider>
  );
}

function Toolbar() {
  return <ThemeButton />;
}

function ThemeButton() {
  const { theme, setTheme } = useContext(ThemeContext);
  return (
    <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
      Current Theme: {theme}
    </button>
  );
}

4. Redux

I have hands-on experience implementing Redux for complex applications, managing state predictably and ensuring smooth data flow.

  • Strong understanding of actions & reducers
  • Integration with React components
  • Centralized state management
import { createStore } from 'redux';

const initialState = { count: 0 };

function reducer(state = initialState, action) {
  switch (action.type) {
    case 'INCREMENT': return { count: state.count + 1 };
    case 'DECREMENT': return { count: state.count - 1 };
    default: return state;
  }
}

const store = createStore(reducer);
export default store;
